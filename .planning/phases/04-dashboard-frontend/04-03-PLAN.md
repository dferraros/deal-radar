State: `const [isWatchlisted, setIsWatchlisted] = useState(initialState)` and `const [loading, setLoading] = useState(false)`.

Toggle logic:
```typescript
const handleToggle = async () => {
  const next = !isWatchlisted;
  setIsWatchlisted(next);   // optimistic
  setLoading(true);
  try {
    if (next) {
      await axios.post('/api/watchlist', { company_id: companyId });
    } else {
      // Need watchlist item ID to DELETE. GET /api/watchlist to find it.
      const res = await axios.get('/api/watchlist');
      const item = res.data.find((w: { company_id: string; id: string }) => w.company_id === companyId);
      if (item) await axios.delete(`/api/watchlist/${item.id}`);
    }
    onToggle?.(next);
  } catch {
    setIsWatchlisted(!next);   // revert on error
  } finally {
    setLoading(false);
  }
};
```

Render:
```tsx
<Button
  variant="secondary"
  disabled={loading}
  onClick={handleToggle}
  className={isWatchlisted ? 'border-amber-400/50 text-amber-400' : ''}
  icon={loading ? undefined : (isWatchlisted ? StarFilled : StarOutline)}
>
  {loading ? 'Saving...' : (isWatchlisted ? 'On Watchlist' : 'Add to Watchlist')}
</Button>
```

```typescript
const save = async () => {
  setEditing(false);
  if (draft === note) return;   // no change
  const prev = note;
  setNote(draft);               // optimistic
  try {
    await axios.put(`/api/watchlist/${watchlistItemId}/notes`, { notes: draft });
    onSave?.(draft);
  } catch {
    setNote(prev);              // revert
    setError('Failed to save note. Try again.');
    setTimeout(() => setError(null), 3000);
  }
};
```

```tsx
{error && <span className="text-xs text-red-400 block mt-0.5">{error}</span>}
```
